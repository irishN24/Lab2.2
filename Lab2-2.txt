#include <stdio.h>
#include <math.h>
#define PI 3.14159265358979323846 
int main()
{
	double Y, a, b, h;
	printf("Input a, b, h:\n");
	double rez = scanf("%lf%lf%lf", &a, &b, &h);
	if (rez == 3.000000000000000000) 
	{
		for (double x = a * PI; x <= b * PI;x += h * PI )
		{
			Y = cos(x);
			printf("x: %lf,Y: %lf\n",x, Y);
			
		}
	}
	else
	{
		printf("Wrong Data");
	}
	return 0;
}
